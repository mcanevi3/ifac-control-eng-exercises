\begin{tikzpicture}
  [
  declare function={
    myresp(\x) = (exp(-4*(\x))/(16 + (\wnone)^2 - 8*(\wnone)*(\zetaone))
     - (exp(-(\x)*(\wnone)*(\zetaone)) *
        (cosh((\x)*(\wnone)*sqrt(-1+(\zetaone)^2))
         + (sinh((\x)*(\wnone)*sqrt(-1+(\zetaone)^2))
            * (-4 + (\wnone)*(\zetaone))
           )/((\wnone)*sqrt((\zetaone)^2 - 1))
        )
       ) / (16 + (\wnone)^2 - 8*(\wnone)*(\zetaone))
    );
  }
]

  xticklabel style={
      /pgf/number format/fixed,

  },

        /pgf/number format/.try,
        /pgf/number format/fixed,
        /pgf/number format/fixed zerofill,
        /pgf/number format/precision=2 % <-- Sets two decimal places
  },
]

\end{axis}
