\begin{tikzpicture}
  [
  declare function = {
    sindeg(\x)=sin(deg(\x));
    cosdeg(\x)=cos(deg(\x));
    myresp2(\x) =
      (
        exp(-4*\x)
        - exp(-\wntwo*\zetatwo*\x)
          * (
              cosdeg(\wdtwo*\x)+(
              sindeg(\wdtwo*\x)*(-4+\wntwo*\zetatwo)/(\wdtwo) )
            )
      )
      / ( 16 + \wntwo^2 - 8*\wntwo*\zetatwo );
  }
]

\begin{axis}[
  width=0.8\columnwidth,
  % height=0.4\columnwidth,
  xlabel={$t$},
  ylabel={$y(t)$},
  grid=both,
  domain=0:\tfintwo,
  samples=200,
  thick,
  xticklabel style={
      /pgf/number format/fixed,
      /pgf/number format/precision=1
  },
  scaled ticks=false,
  yticklabel style={
        /pgf/number format/.try,
        /pgf/number format/fixed,
        /pgf/number format/fixed zerofill,
        /pgf/number format/precision=2 % <-- Sets two decimal places
  },
]
\addplot[blue]{myresp2(x)};
\end{axis}
\end{tikzpicture}
