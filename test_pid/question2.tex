\pgfmathsetmacro{\zetaone}{2+(5-2)*(10*\xone+\xzero)/99}
\pgfmathsetmacro{\wnone}{4+(1-4)*(10*\xone+\xzero)/99}
\pgfmathsetmacro{\tfinone}{max(2*4/(\wnone*(\zetaone-sqrt(\zetaone^2-1))), 2*1)}

\pgfmathsetmacro{\zetatwo}{0.1+(0.4-0.1)*(10*\xone+\xzero)/99}
\pgfmathsetmacro{\wntwo}{0.5+(1-0.5)*(10*\xone+\xzero)/99}
\pgfmathsetmacro{\tfintwo}{max(1.5*4/(\wntwo*\zetatwo), 2*1)}
\pgfmathsetmacro{\wdtwo}{\wntwo*sqrt(1-\zetatwo^2)}


\question[50] Which of the following does not overshoot?
\begin{randomizechoices}
\CorrectChoice \mbox{ \begin{tikzpicture}
  [
  declare function={
    myresp(\x) = (exp(-4*(\x))/(16 + (\wnone)^2 - 8*(\wnone)*(\zetaone))
     - (exp(-(\x)*(\wnone)*(\zetaone)) *
        (cosh((\x)*(\wnone)*sqrt(-1+(\zetaone)^2))
         + (sinh((\x)*(\wnone)*sqrt(-1+(\zetaone)^2))
            * (-4 + (\wnone)*(\zetaone))
           )/((\wnone)*sqrt((\zetaone)^2 - 1))
        )
       ) / (16 + (\wnone)^2 - 8*(\wnone)*(\zetaone))
    );
  }
]

\begin{axis}[
  xlabel={$t$},
  ylabel={$y(t)$},
  grid=both,
  domain=0:\tfinone,
  samples=100,
  thick
]
\addplot[blue]{myresp(x)};
\addlegendentry{Response}
\end{axis}
\end{tikzpicture}
}
\choice  \mbox{ \begin{tikzpicture}
  [
  declare function = {
    sindeg(\x)=sin(deg(\x));
    cosdeg(\x)=cos(deg(\x));
    myresp2(\x) =
      (
        exp(-4*\x)
        - exp(-\wntwo*\zetatwo*\x)
          * (
              cosdeg(\wdtwo*\x)+(
              sindeg(\wdtwo*\x)*(-4+\wntwo*\zetatwo)/(\wdtwo) )
            )
      )
      / ( 16 + \wntwo^2 - 8*\wntwo*\zetatwo );
  }
]

\begin{axis}[
  xlabel={$t$},
  ylabel={$y(t)$},
  grid=both,
  domain=0:\tfintwo,
  samples=100,
  thick
]
\addplot[blue]{myresp2(x)};
\addlegendentry{Response}
\end{axis}
\end{tikzpicture}
}

\end{randomizechoices}
